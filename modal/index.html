# Modal 模态窗

演示一个使用外部依赖的组件，打包只会包含组件代码，无法独立使用。

# 安装

npm i @palvue/modal

# 引入

import Modal from '@palvue/modal'
Vue.use(Modal)

# 使用

# 代码演示


<template>
  <a-button @click="show = true">点击打开模态窗</a-button>
  <zk-modal v-model="show" />
</template>

<script>
export default {
  data() {
    return {
      show: false,
    }
  },
  mounted() {},
  methods: {},
}
</script>
